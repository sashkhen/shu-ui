## Install

```sh
npm install @sashkhen/shu-ui
```

## Usage

```tsx
import { Button } from "shu-ui";

const Example = () => {
  return (
    <Button
      variant="solid"
      size="medium"
      onClick={() => console.log("Clicked")}
    >
      Button
    </Button>
  );
};

export default Example;
```
